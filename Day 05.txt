Each admission costs $50 for patients without insurance, and $10 for patients with insurance. All patients with an even patient_id have insurance.

Give each patient a 'Yes' if they have insurance, and a 'No' if they don't have insurance. Add up the admission_total cost for each has_insurance group.


select has_insurance,

case 
when has_insurance='Yes' then frequency *10
when has_insurance='No' then frequency *50
end as cost_after_insurance

from 

(select 
has_insurance, count (*) as Frequency
from

(
select *,
case 
  when patient_id % 2 = 0 then 'Yes'
  when patient_id % 2 = 1 then 'No'
end as has_insurance from admissions)
as frequency group by has_insurance)
